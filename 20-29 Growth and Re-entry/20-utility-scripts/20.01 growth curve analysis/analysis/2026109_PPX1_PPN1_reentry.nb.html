my_diag_plot <- function(df, df.sum, well){
  # df is the data frame containing the data
  # df.sum contains the per well estimates calculated from the raw data
  # this function is aimed at one well of data, specified by "well"
  # check to see if the well number is valid
  if(!well %in% df$Well){
    stop(paste(well, "is not a valid well", sep = " "))
  }
  
  # subset the df
  dat <- filter(df, Well == well)
  # extract the estimates
  par <- filter(df.sum, Well == well)
  
  # for overlaying the derivatives on the plot,
  # we will calculate a scaling factor
  tmp <- dat %>% summarize(
    min_OD = min_gc(Measurements), max_OD = max_gc(Measurements),
    min_deriv = min_gc(Deriv), max_deriv = max_gc(Deriv),
    rng_OD = max_OD - min_OD,
    rng_deriv = max_deriv - min_deriv,
    sf_deriv = rng_OD / rng_deriv,
    min_logOD = min_gc(log(Measurements)), max_logOD = max_gc(log(Measurements)),
    min_percap_deriv = min_gc(percap_deriv), max_percap_deriv = max_gc(percap_deriv),
    rng_logOD = max_logOD - min_logOD,
    rng_percap_deriv = max_percap_deriv - min_percap_deriv,
    sf_percap_deriv = rng_logOD / rng_percap_deriv
  )
  
  # make the linear growth curve plot
  p1 <- dat %>% 
    mutate(
      scaled_deriv = (Deriv - tmp$min_deriv)*tmp$sf_deriv + tmp$min_OD
    ) %>% 
    ggplot(aes(x = Time, y = Measurements)) + geom_point(size = 0.4) + 
    geom_line() +
    geom_line(aes(x = Time, y = scaled_deriv), color = "blue", linetype = 2) +
    ylab("OD")
    
  # make the semi-log plot
  p2 <- dat %>% 
    mutate(
      scaled_percap_deriv = (percap_deriv-tmp$min_percap_deriv)*tmp$sf_percap_deriv + 
        tmp$min_logOD
    ) %>% 
    ggplot(aes(x = Time, y = log(Measurements))) + geom_point() + geom_line() +
    geom_line(aes(x = Time, y = scaled_percap_deriv),
              color = "blue", linetype = 2) +
    geom_abline(
      data = par,
      color = "red", 
      aes(slope = max_percap,
          intercept = log(max_percap_dens) - max_percap * max_percap_time)
    ) +
    geom_vline(data = par,
               aes(xintercept = lag_time), lty = 2) +
    geom_hline(data = par,
               aes(yintercept = log(min_dens))) +
    ylab("log OD")
  # draw the derivative and per_capita_derivative

  plot_grid(p1, p2, nrow = 2)
}
